using System;

namespace StripedTowel
{
    class Towel
    {
        static void Main()
        {
            int width = int.Parse(Console.ReadLine());

            for (int i = 0; i < (int)(width * 1.5); i++)
            {
                Console.WriteLine(new string('x', 51).Replace("x", "#..").Substring(i % 3, width));             
            }
        }
    }
}
